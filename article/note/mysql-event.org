#+TITLE: mysql 计划任务

* 开启计划任务
#+BEGIN_SRC sql
SET GLOBAL event_scheduler="ON" 
#+END_SRC

* 关闭计划任务
#+BEGIN_SRC sql
SET GLOBAL event_scheduler="OFF" 
#+END_SRC

* 定时任务
#+BEGIN_SRC sql
create EVENT event_name
ON SCHEDULE EVERY 1 SECOND STARTS '2016-11-30 14:13:35' DO 
INSERT INTO test.t00 (`time`) VALUES (unix_timestamp());
#+END_SRC
